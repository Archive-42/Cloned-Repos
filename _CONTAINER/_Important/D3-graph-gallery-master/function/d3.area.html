The d3.area() function




The d3 library provides helper functions to draw elements more easily. Here is a description of the d3.area() function.


Histogram section

It is possible to draw a line in pure svg like in the following example:

<!DOCTYPE html>
<meta charset="utf-8">

<!-- Load d3.js -->
<script src="https://d3js.org/d3.v4.js"></script>

<!-- Create a div where the graph will take place -->
<svg width=1000 height=110></svg>

<script>

// Draw the line segment per segment
d3.select("svg")
  .append("path")
    .attr("d", "M100 100 L100 30 L200 50 L300 90 L400 20 L500 10 L600 60 L700 50 L800 30 L900 50 L900 100 Z")
    .attr("stroke", "black")
    .attr("fill", "#69b3a2")

</script>

It is not very convenient, thus d3.js offer helper function that help to do that more easily

← Edit me! 
<!DOCTYPE html>
<meta charset="utf-8">

<!-- Load d3.js -->
<script src="https://d3js.org/d3.v4.js"></script>

<!-- Create a div where the graph will take place -->
<svg width=1000 height=110 id="my_dataviz"></svg>
← Edit me! 
<script>

// The data
var data = [ {x: 100, y: 30}, {x: 200, y: 10}, {x: 300, y: 90}, {x: 400, y: 20}, {x: 500, y: 80}, {x: 600, y: 20}, {x: 700, y: 90}, {x: 800, y: 20}, {x: 900, y: 29}]

// Draw the line segment per segment
d3.select("#my_dataviz")
  .append("path")
  .datum(data)
    .attr("d", d3.area()
      .x(function(d) { return d.x })
      .y1(function(d) { return d.y })
      .y0(100)
    )
  .attr("fill", "#69a3b2")
  .attr("stroke", "black")


</script>
